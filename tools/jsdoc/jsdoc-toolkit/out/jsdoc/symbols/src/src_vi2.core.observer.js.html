164 	addWidget : function(obj){ 
165 		var _this = this;   	
166 		obj.player = this.player; 
167 		this.clock.addHook(obj.name, obj);	
168 
169 		if(obj.type === 'annotation'){   
170 			obj.appendToDOM( this.current_stream ); // former: this.options.id
171 			$(this.player).bind('annotation.begin.'+obj.name, function(e, a, b){ obj.begin(e, a, b);});
172 			$(this.player).bind('annotation.end.'+obj.name, function(e, a){ obj.end(e, a);});
173 		}	
174 alert(3)
175 		// xxx needs to be put into widgets
176 		switch(obj.name){
177 			case 'relatedVideos' :
178 				$(this.player).bind('video.end', function(e, a){ obj.showLinkSummary(); });
179 				break;  
180 			case 'log' :
181 				$(this.player).bind('log', function(e, msg){ obj.add(msg); });
182 				break;
183 		}
184 	
185 		// register widget	
186 		this.widget_list[obj.name] = obj;   
187 		return true;
