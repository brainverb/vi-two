 11 	var TOC = $.inherit(Annotation, /** @lends TableOfContents# */{ // 
 12 
 13 		/** @constructs
 14 		*		@extends Annotation
 15 		*		@param {object} options An object containing the parameters
 16 		*/
 17   	__constructor : function(options) {
 18   			this.options = $.extend(this.options, options);  
 19 		},
 20 		
 21 		name : 'toc',
 22 		type : 'annotation',
 23 		options : {
 24 			selector: '#toc', 
 25 			timelineSelector : '.vi2-video-seek',
 26 			hasTimelineMarker: true, 
 27 			hasMenu : true,
 28 			path:'/'
 29 		},
 30 		currentTocElement : 0,
 31 		elements : [],
 32 
 33 		/**
 34 		Initializes the table of content
 35 		*/
 36 		init : function(annotations){
 37 			var _this = this;
 38 			// prepare toc data
 39 			var e = {}; e.tags = []; e.tags.occ = [];
 40 			$.each(annotations, function(i, val){
 41 				if(val.type == 'toc'){ 
 42 					if( val.note !== "missing"){
 43 						_this.elements.push(val.t1);
 44 					}	
 45 					e.tags.push({name: val.title, occ:[val.t1]});
 46 				}
 47 			});
 48 			// 
 49 			if( this.options.hasMenu ){
 50 				this.buildMenu(e.tags)
 51 			}	
 52 			// 
 53 			if( this.options.hasTimelineMarker ){
 54 				this.buildTimelineMarkers(e);
 55 			}
 56 					
 57 			// update toc highlight on time update
 58 			vi2.observer.player.video.addEventListener('timeupdate', function(e) { 
 59 				// reset highlight
 60 				//		$(_this.options.selector+' li').each(function(i, val){ $(this).removeClass('toc-hightlight');})
 61 				// highlight toc entry
 62 				//		$(_this.options.selector+ ' li#t'+this.formatTime(obj.content.target)).addClass('toc-highlight');
 63 			});
 64 		},		
 65 		
 66 		
 67 		/**
 68 		Loads toc date from database in order to generate corresponding DOM elements.
 69 		@ Resulting format of DOM element: <div type="toc" starttime=83 duration=1 id="">Objectives of the lecture</div>
 70 		*/
 71 		appendToDOM : function(id){ 
 72 			var _this = this; 
 73 			$(vi2.dom).find('[type="toc"]').each(function(i,val){ $(this).remove(); });
 74 			$.each(	vi2.db.getTocById(id), function( i, val ){  
 75 				var toc = $('<div></div>')
 76 					.attr('type',"toc")
 77 					.attr('note', val.note )
 78 					.attr('starttime', val.start)
 79 					.attr('duration', 10)
 80 					.attr('author', 'none')
 81 					.attr('id', "")
 82 					.text( decodeURIComponent(val.label) )
 83 					.appendTo( vi2.dom )
 84 					;
 85 			});
 86 		},					
 87 				
 88 		/** 
 89 		During the playback corresponds with a toc entry, the entry gets highlighted in the menu.	
 90 		*/
 91 		begin : function(e, id, obj){ 
 92 				// reset highlight
 93 				$(this.options.selector+' li').each(function(i, val){ $(this).removeClass('toc-highlight') ;})
 94 				// highlight toc entry
 95 				$('.toc-'+obj.displayPosition.t1).addClass('toc-highlight');
 96 			
 97 		},
 98 	
 99 		/** 
100 		
101 		*/
102 		end : function(e, id, obj){ 
103 			//$('.toc-'+obj.displayPosition.t1).removeClass('toc-highlight');
104 		},
105 		
106 		
107 		/**
108 		
109 		*/
110 		buildMenu : function(tocData){
111 			var _this = this;
112 			var toc = $('<ul></ul>').addClass('toc-list'); 
113 			
114 			$.each(tocData, function(i, val){
115 					var a = $('<a></a>')
116 					.text( val.name )
117 					.addClass('id-'+ val.occ[0])
118 					.attr('href', '#!t=npt:' + val.occ[0] + '') // former: main.options.id
119 					;				
120 					var li = $('<li></li>')
121 						.addClass('toc-'+val.occ[0])
122 						.attr('id', ''+ i)
123 						//.css('list-style-image',  "url('"+_this.options.path+"user-"+val.author+".png')")
124 						.html(a)
125 						.appendTo( toc )
126 						; 
127 					if( val.note === "missing"){
128 						li.addClass('toc-disabled');
129 					}else{
130 						li.click(function(){
131 							vi2.observer.log('clicktocfromlist:'+val.name.replace(/,/g,'##') +' '+val.author+' '+ val.occ[0]); 
132 							vi2.observer.player.currentTime( val.occ[0] );
133 							_this.currentTocElement = i;
134 						});	
135 					}	
136 			});
137 			
138 			
139 			// sort list entries by time and append them
140 			toc.find('li').tsort({attr:"id"});  // tsort is error prune under chromium
141 			$(_this.options.selector).html(toc);		
142 			
143 		},
144 		
145 		
146 		/** 
147 		Displays table of contents as markers on the timeline
148 		*/
149 		buildTimelineMarkers : function(e){  
150 			var _this= this; 
151 				// display toc occurence on timeline to motivate further selection
152 				var f = function(_left, _name){ 
153 					return sp = $('<span></span>')					
154 						.addClass('toc-timeline-marker ttoc')
155 						.attr('style','left:'+_left+'px;')
156 						.tooltip({delay: 0, showURL: false, bodyHandler: function() { return $('<span></span>').text(_name);} });
157 				};
158 								
159 				$.each(e.tags, function(){ 
160 					var progress = this.occ[0] / vi2.observer.player.duration();
161 					progress = ((progress) * $(_this.options.timelineSelector).width());
162   	    	if (isNaN(progress) || progress > $(_this.options.timelineSelector).width()) { return;}
163 	 				$(_this.options.timelineSelector).append(f(progress, this.name));
164  				});
165 		},
166 		
167 		
168 		/** 
169 		Jumps to the next element of the table of content 
170 		*/
171 		nextElement : function(){
172 			//$('.toc-'+ (this.currentTocElement + 1) ).click();
173 			this.currentTocElement = this.currentTocElement < this.elements.length ? this.currentTocElement + 1 : this.currentTocElement;
174 			vi2.observer.player.currentTime( this.elements[ this.currentTocElement ] );
175 			//
176 		},
177 		
178 		
179 		/** 
180 		Jumps to the previous element of the table of content. 
181 		*/
182 		previousElement : function(){
183 			this.currentTocElement = this.currentTocElement == 0 ? this.currentTocElement : this.currentTocElement -1 ;
184 			vi2.observer.player.currentTime( this.elements[ this.currentTocElement ] );
185 		}
