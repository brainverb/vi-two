 10 var Vi2_PlaybackSpeed = $.inherit(/** @lends PlaybackSpeed# */{
 11 
 12 	/** 
 13 	*		@constructs 
 14 	*		
 15 	*/
 16 	__constructor : function(options, observer) { 
 17 			this.options = $.extend(this.options, options);
 18 			this.observer = observer;
 19 			this.player = this.observer.player;
 20 			this.video = document.getElementById( this.options.video_selector ); 
 21 			this.init();
 22 	},
 23 	
 24 	name : 'playbackSpeed',
 25 	options : {
 26 		video_selector : 'video1',
 27 		selector: '.control-bar',
 28 		speed_steps: [0.3,0.5,0.8,1.0,1.5,2.0,3.0,4.0]	
 29 	},
 30 	speed : 1,
 31 	observer : '',
 32 	player : '',
 33 	video : '',	
 34 	speedIndex : 3,
 35 	
 36 
 37 	/* Initializes the playback speed controls */
 38 	init : function(selector){ 
 39 		var _this = this;
 40 		var container = $('<div></div>')
 41 			.append($('<div></div>').text('1.0x').addClass('speed-label'))
 42 			.addClass('speed-controls')
 43 			.bind('mouseenter', function(e){
 44 				$('.speed-controls > ul').css('display','block');
 45 			})
 46 			.bind('mouseleave', function(e){
 47 				$('.speed-controls > ul').css('display','none');
 48 			})
 49 			.tooltip({
 50 				delay: 0, 
 51 				showURL: false, 
 52 				bodyHandler: function() { 
 53 					return $('<span></span>')
 54 						.text('Wiedergabegeschwindigkeit');
 55 				} 
 56 			})
 57 			.appendTo( this.options.selector );
 58 			
 59 		var options = $('<ul></ul>')
 60 			.addClass('select-speed')
 61 			.appendTo(container);
 62 		
 63 		$.each( this.options.speed_steps, function(i, val){ 
 64 			var sel = $('<li></li>')
 65 				.attr('speed', val)
 66 				.text(val+'x')
 67 				.click(function(e){
 68 					_this.setCurrentSpeed( $(this).attr('speed') );
 69 				})
 70 				.appendTo(options);
 71 				
 72 		})
 73 	},
 74 	
 75 	/* */
 76 	displaySpeed : function(){
 77 		// need to be implemented
 78 	},
 79 	
 80 	
 81 	
 82 	/**** INTERFACES **********/
 83 	
 84 	/* Returns the current playback speed*/
 85 	getCurrentSpeed : function(){
 86 		return this.speed;
 87 	},
 88 	
 89 	/* Sets the playback speed */ 
 90 	setCurrentSpeed : function(speed){ 
 91 		if( this.options.speed_steps.indexOf( parseFloat(speed) ) != -1){
 92 			// set speed
 93 			this.video.defaultPlaybackRate = 1.0; 
 94 			this.video.playbackRate = speed; 
 95 			this.speed = speed;
 96 			this.speedIndex = this.options.speed_steps.indexOf( parseFloat(speed) );
 97 			// set label
 98 			$('.speed-label').text( speed + 'x');
 99 			// close select menu
100 			$('.speed-controls > ul').css('display','none');
101 			// log it
102 			vi2.observer.log( this.url + ' change_speed: ' + this.player.currentTime() + ' speed: ' + speed);
103 		}	
104 	},
105 	
106 	/* Increases the playback speed by one step in the index of the given values */
107 	increaseSpeed : function(){ 
108 		if( this.speedIndex < this.options.speed_steps.length ){
109 			this.setCurrentSpeed( this.options.speed_steps[ this.speedIndex + 1 ] );
110 		}	
111 	},
112 	
113 	/* Decreases the playback speed by one step in the index of the given values */
114 	decreaseSpeed : function(){ 
115 		if( this.speedIndex > 1 ){
116 			this.setCurrentSpeed( this.options.speed_steps[ this.speedIndex -1 ] );
117 		}	
118 	}
119 	
120 	
121 }); // end class
